Deep dive into asynchronous JavaScript with practical examples and best practices for handling promises.

// Async/await example
async function fetchUserData(userId) {
  try {
    const response = await fetch(`/api/users/${userId}`);
    const user = await response.json();
    return user;
  } catch (error) {
    console.error('Error:', error);
  }
}
